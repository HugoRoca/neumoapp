docker-compose up -d

# 2. Crear entorno virtual e instalar
python3 -m venv venv
source venv/bin/activate
pip install -r requirements.txt

# 3. Inicializar base de datos con datos de prueba
python init_db.py

# 4. Ejecutar la API
uvicorn main:app --reloads
